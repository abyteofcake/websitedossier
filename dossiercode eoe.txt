<script>
// Attach event listeners to tabs
document.querySelectorAll('.tab-group button.tab').forEach((tab, index) => {
  tab.addEventListener("click", () => handleTab(index));
});

function handleTab(index) {
  // Remove active class from all tabs and articles
  document.querySelectorAll('.tab-group button.tab, .tab-group article.tab-content').forEach(node => {
    node.classList.remove('active');
  });
  
  // Add active class to the clicked tab and corresponding article
  document.querySelectorAll('.tab-group button.tab')[index].classList.add('active');
  document.querySelectorAll('.tab-group article.tab-content')[index].classList.add('active');
}
</script>
